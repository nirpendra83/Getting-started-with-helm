Step-by-step setup for Angular Frontend + .NET Core Backend + PostgreSQL using Docker Compose

1. Create root project folder

mkdir FullStackApp && cd FullStackApp

2. Create subdirectories

mkdir frontend backend postgres-data

3. Create Angular frontend project (you must have Angular CLI installed)

cd frontend ng new frontend-app –routing=false –style=css cd ..

4. Create .NET Core Web API backend (you must have .NET SDK installed)

cd backend dotnet new webapi -n BackendApi cd BackendApi

Add a sample controller (HelloWorldController)

mkdir Controllers cat «EOF > Controllers/HelloWorldController.cs using Microsoft.AspNetCore.Mvc;

namespace BackendApi.Controllers { [ApiController] [Route(“api/[controller]”)] public class HelloWorldController : ControllerBase { [HttpGet] public IActionResult Get() { return Ok(“Hello from .NET Core API!”); } } } EOF

5. Add Dockerfile to backend

cat «EOF > Dockerfile FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS base WORKDIR /app

FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build WORKDIR /src COPY . . RUN dotnet restore RUN dotnet publish -c Release -o /app/publish

FROM base AS final WORKDIR /app COPY –from=build /app/publish . ENTRYPOINT [“dotnet”, “BackendApi.dll”] EOF

cd ../../

6. Create docker-compose.yml in root folder

cat «EOF > docker-compose.yml version: ‘3.8’

services: postgres: image: postgres:15 container_name: postgres environment: POSTGRES_USER: admin POSTGRES_PASSWORD: admin123 POSTGRES_DB: sampledb volumes: - ./postgres-data:/var/lib/postgresql/data ports: - “5432:5432”

api: build: context: ./backend/BackendApi container_name: backend-api ports: - “5000:80” depends_on: - postgres

frontend: image: node:18 working_dir: /app volumes: - ./frontend/frontend-app:/app command: bash -c “npm install && ng serve –host 0.0.0.0 –port 4200” ports: - “4200:4200” depends_on: - api EOF

7. Start the entire stack

Run this from FullStackApp root directory

docker-compose up –build -d
